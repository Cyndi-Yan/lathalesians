% SDE = stochastic differential equation
\nomenclature[000]{iff}{if and only if}
\nomenclature[001]{$A, B, \ldots$}{usually denote sets}
\nomenclature[002]{$\mathcal{A}, \mathcal{B}, \ldots$}{usually denote collections of sets}
\nomenclature[003]{$\emptyset$}{the empty set}
\nomenclature[004]{$\setcomplement{A}$}{the complement of a set $A$}
\nomenclature[005]{$A \times B$}{the Cartesian product of the sets $A$ and $B$}
\nomenclature[006]{$\N$}{the natural numbers, $1, 2, 3, \ldots$}
\nomenclature[007]{$\R$}{the real line, $(-\infty, +\infty)$}
\nomenclature[008]{$\Rp$}{the positive real numbers, $(0, +\infty)$}
\nomenclature[009]{$\Rnn$}{the nonnegative real numbers, $[0, +\infty)$}
\nomenclature[010]{$\Rn$}{the negative real numbers, $(-\infty, 0)$}
\nomenclature[011]{$\Rx$}{the affinely extended real line, $[-\infty, +\infty]$}
\nomenclature[012]{$f^{-1}(A)$}{the inverse image of the set $A$ under the function $f$}
\nomenclature[013]{$x_{1:n}$}{$\transpose{\begin{pmatrix} x_1 & x_2 & \ldots & x_n \end{pmatrix}}$}
\nomenclature[014]{$\sigma(\mathcal{A})$}{the $\sigma$-algebra generated by the family of sets $\mathcal{A}$}
\nomenclature[015]{$\mu$-a.e.}{$\mu$-almost everywhere, for some measure $\mu$}
\nomenclature[016]{$\mu$-a.a.}{$\mu$-almost all}
\nomenclature[017]{$\mu$-a.s.}{$\mu$-almost surely}
\nomenclature[018]{i.i.d.}{independent and identically distributed}
\nomenclature[019]{$p(\cdot \given \cdot)$}{a generic conditional probability density or mass function, with the arguments making it clear which conditional distribution it relates to}
\nomenclature[020]{$\mathbb{T}$}{the time set}
\nomenclature[021]{$X$}{the signal process}
\nomenclature[022]{$\mathbb{S}$}{the state space of the signal process, assumed to be a complete separable metric space}
\nomenclature[023]{$\mathcal{B}(\mathbb{S})$}{the Borel $\sigma$-algebra of $\mathbb{S}$}
\nomenclature[024]{$\collection{\mathcal{X}}_{t \in \mathbb{T}}$}{the usual augmentation of the filtration generated by $X$}
